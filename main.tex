\section{文献综述与理论基础}

\begin{frame}{文献综述}{从均值方差到风险预算}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item 均值方差模型强调收益与风险权衡，但对预期收益估计敏感。
        \item 风险平价将关注点转向风险贡献，弱化对收益预测的依赖。
        \item 宽松风险平价可被视为在严格风险预算和实际投资约束之间的折中。
      \end{itemize}
    \end{column}
    \begin{column}{0.4\textwidth}
      \begin{block}{理论脉络}
        均值方差模型\\
        $\Downarrow$\\
        最小方差与分散化\\
        $\Downarrow$\\
        风险平价与风险预算\\
        $\Downarrow$\\
        宽松风险平价
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{理论基础}{风险贡献与预算约束}
  设组合权重为 $w$，协方差矩阵为 $\Sigma$，组合波动率为
  \[
    \sigma_p = \sqrt{w^\top \Sigma w}.
  \]
  第 $i$ 个资产的边际风险贡献与总风险贡献可写为
  \[
    \mathrm{MRC}_i = \frac{(\Sigma w)_i}{\sigma_p}, \qquad
    \mathrm{RC}_i = w_i \cdot \mathrm{MRC}_i.
  \]
  \begin{block}{宽松风险预算思想}
    严格风险平价要求 $\mathrm{RC}_i$ 接近目标预算 $b_i\sigma_p$；宽松版本允许风险贡献在上下界内波动，以吸收估计误差和市场状态变化。
  \end{block}
\end{frame}

\section{研究方法与模型设计}

\begin{frame}{模型设计}{宽松风险平价的优化问题}
  \begin{equation}
    \begin{aligned}
      \min_{w}\quad & w^\top \Sigma w \\
      \mathrm{s.t.}\quad
      & \sum_{i=1}^{N} w_i = 1,\quad 0 \leq w_i \leq u_i,\\
      & \ell_i \leq \frac{\mathrm{RC}_i}{\sigma_p} \leq h_i,\quad i=1,\ldots,N.
    \end{aligned}
  \end{equation}
  \begin{itemize}
    \item $\ell_i$ 与 $h_i$ 分别表示第 $i$ 类资产风险预算比例的下界和上界。
    \item 若上下界收窄，模型接近传统风险平价；若上下界放宽，模型更接近约束最小方差。
  \end{itemize}
\end{frame}

\begin{frame}{结果讨论}{如何避免过度解释}
  \begin{itemize}
    \item 若宽松风险平价在某些窗口表现更好，应说明其可能来自回撤控制、波动管理或分散化改善，而非声称策略始终占优。
    \item 若交易成本后指标下降，应保留该结果，并讨论月度再平衡下换手率对实际可投资性的影响。
    \item 若协方差估计方法改变结论，应将其作为模型不确定性的一部分，而不是事后选择最优口径。
  \end{itemize}
\end{frame}

\section{主要结论与不足}

\begin{frame}{主要结论}{答辩中的收束表达}
  \begin{enumerate}
    \item 本文围绕传统风险平价在现实市场中的约束，构建了宽松风险预算框架。
    \item 在全球 ETF 资产池和月度再平衡设定下，模板示例展示了收益、风险和回撤指标的比较方式。
    \item 稳健性部分从参数区间、协方差估计和交易成本三个角度检验结论敏感性。
  \end{enumerate}
  \begin{block}{答辩建议}
    结论页应回扣研究问题、方法和证据，不宜新增复杂结果。
  \end{block}
\end{frame}

\begin{frame}{研究不足与展望}
  \begin{itemize}
    \item 数据范围方面：ETF 上市时间和资产类别覆盖会影响样本长度与可比性。
    \item 模型方面：协方差估计误差仍可能传导至权重，未来可进一步比较贝叶斯或动态相关模型。
    \item 实践方面：真实交易还需考虑税费、滑点、流动性容量和再平衡执行偏差。
  \end{itemize}
  \begin{alertblock}{展望}
    后续研究可将宽松风险预算与宏观状态识别、尾部风险控制或生命周期配置需求结合。
  \end{alertblock}
\end{frame}
